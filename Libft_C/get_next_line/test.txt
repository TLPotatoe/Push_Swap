#include "get_next_line_bonus.h"
#include "stdio.h"

int	main(void)
{
	int		fd;
	char	*line;

	fd = open("file.txt", O_RDONLY);
	while (1)
	{
		line = get_next_line(fd);
		if (!line)
			break ;
		printf("->%s", line);
		free(line);
	}
	close(fd);
}
